"); console.log("Welcome " + userName) console.log("Let's play a quiz how well do you know me?") function play(question, answer) { var userAnswer = readlineSync.question(question); if (userAnswer === answer) { console.log("right!"); score = score + 1; } else { console.log("wrong!"); score = score - 1; } console.log("current score: ", score); console.log("-------------") } var questionone = { question: "Where do I live?
